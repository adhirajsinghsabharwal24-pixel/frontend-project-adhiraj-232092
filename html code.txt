  <script>
    /* ------------------------------
       Sample product data (replace if needed)

    const products = [
      {id:1,title:'Crisp White Tee',desc:'Slim cotton tee — front seam detail',price:799,cat:'tops',img:'https://images.unsplash.com/photo-1520975910394-9a3b7a2b2f23?q=80&w=1200&auto=format&fit=crop&s=3d3b2c1a2f9a'},

      {id:3,title:'Lightweight Shell',desc:'Tech shell with minimal zipper',price:3499,cat:'outer',img:'https://images.unsplash.com/photo-1541099649105-7a1b2c3d4e5f?q=80&w=1200&auto=format&fit=crop&s=2b3c4d5e6f7a'},
      {id:4,title:'Striped Shirt',desc:'Button-down with chest detail',price:1199,cat:'tops',img:'https://images.unsplash.com/photo-1520975927108-9b1497a2d2a4?q=80&w=1200&auto=format&fit=crop&s=4a5b6c7d8e9f'},
      {id:5,title:'Casual Shorts',desc:'Stretch cotton — campus friendly',price:999,cat:'bottoms',img:'https://images.unsplash.com/photo-1503342452485-86d2e9c3f1af?q=80&w=1200&auto=format&fit=crop&s=5b6c7d8e9f0a'},
      {id:6,title:'Minimal Cap',desc:'Embossed logo — adjustable',price:499,cat:'accessories',img:'https://images.unsplash.com/photo-1512436991641-6745cdb1723f?q=80&w=1200&auto=format&fit=crop&s=6a7b8c9d0e1f'}
    ];

    /* ------------------------------
       Render grid
       ------------------------------ */
    const grid = document.getElementById('productGrid');
    function render(list){
      grid.innerHTML = '';
      list.forEach(p=>{
        const el = document.createElement('article');
        el.className = 'card';
        el.innerHTML = `
          <div class="thumb" style="background-image:url('${p.img}')" aria-hidden></div>
          <h4 class="p-title">${p.title}</h4>
          <p class="p-desc">${p.desc}</p>
          <div style="display:flex;justify-content:space-between;align-items:center">
            <div class="price">₹${p.price.toLocaleString('en-IN')}</div>
            <div style="display:flex;gap:8px">
              <button class="btn" data-id="${p.id}" data-action="view">Quick</button>
              <button class="btn primary" data-id="${p.id}" data-action="add">Add</button>
            </div>
          </div>
        `;
        grid.appendChild(el);
      });
    }
    render(products);

    /* ------------------------------
       Category filtering
       ------------------------------ */
    document.getElementById('categories').addEventListener('click', e=>{
      const chip = e.target.closest('.chip');
      if(!chip) return;
      document.querySelectorAll('.chip').forEach(c=>c.classList.remove('active'));
      chip.classList.add('active');
      const cat = chip.dataset.cat;
      if(cat === 'all') render(products);
      else render(products.filter(p=>p.cat === cat));
    });

    /* ------------------------------
       Buttons for quick view and add
       ------------------------------ */
    let cart = 0;
    grid.addEventListener('click', e=>{
      const btn = e.target.closest('button');
      if(!btn) return;
      const action = btn.dataset.action;
      const id = Number(btn.dataset.id);
      const product = products.find(p=>p.id===id);
      if(action === 'add'){
        cart++;
        btn.textContent = 'Added ✓';
        setTimeout(()=> btn.textContent = 'Add',900);
      } else if(action === 'view'){
        openModal(product);
      }
    });

    /* ------------------------------
       Modal logic
       ------------------------------ */
    const modal = document.getElementById('modal');
    const modalImg = document.getElementById('modalImg');
    const modalTitle = document.getElementById('modalTitle');
    const modalDesc = document.getElementById('modalDesc');
    const modalPrice = document.getElementById('modalPrice');
    const modalX = document.getElementById('modalX');
    const modalClose = document.getElementById('modalClose');
    const modalAdd = document.getElementById('modalAdd');

    let current = null;
    function openModal(p){
      current = p;
      modalImg.style.backgroundImage = `url('${p.img}')`;
      modalTitle.textContent = p.title;
      modalDesc.textContent = p.desc;
      modalPrice.textContent = '₹' + p.price.toLocaleString('en-IN');
      modal.classList.add('show');
      modal.setAttribute('aria-hidden','false');
    }
    function closeModal(){
      modal.classList.remove('show');
      modal.setAttribute('aria-hidden','true');
    }
    modalX.addEventListener('click', closeModal);
    modalClose.addEventListener('click', closeModal);
    modalAdd.addEventListener('click', ()=>{
      if(!current) return;
      cart++;
      modalAdd.textContent = 'Added ✓';
      setTimeout(()=> modalAdd.textContent = 'Add to cart',900);
    });
    modal.addEventListener('click', e=>{
      if(e.target === modal) closeModal();
    });

    /* ------------------------------
       Animated Typewriter + rotating slogans
       ------------------------------ */
    const typeEl = document.getElementById('typewriter');
    const slogans = [
      "Minimal fronts. Maximum impact.",
      "Designed for campus. Styled for life.",
      "Seams that speak — front-design focus.",
      "Sustainable fabric. Clean finishes.",
      "Timeless silhouettes, student budget friendly."
    ];
    let sIdx = 0;
    let chIdx = 0;
    let deleting = false;
    let delay = 45;

    function typeLoop(){
      const text = slogans[sIdx];
      if(!deleting){
        chIdx++;
        typeEl.textContent = text.slice(0,chIdx);
        if(chIdx === text.length){ deleting = true; setTimeout(typeLoop, 1200); return; }
        setTimeout(typeLoop, delay);
      } else {
        chIdx--;
        typeEl.textContent = text.slice(0,chIdx);
        if(chIdx === 0){ deleting = false; sIdx = (sIdx+1)%slogans.length; setTimeout(typeLoop, 300); return; }
        setTimeout(typeLoop, delay/1.2);
      }
    }
    typeLoop();

    /* ------------------------------
       Optional: Web Speech API for live captions
       ------------------------------ */
    const liveBtn = document.getElementById('liveCapBtn');
    const speechToggle = document.getElementById('speechToggle');
    const micStatus = document.getElementById('micStatus');
    let recognition = null;
    let recognizing = false;

    // feature detect
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition || null;
    if(SpeechRecognition){
      recognition = new SpeechRecognition();
      recognition.lang = 'en-US';
      recognition.interimResults = true;
      recognition.maxAlternatives = 1;
      recognition.continuous = true;

      recognition.onstart = ()=> { recognizing = true; micStatus.textContent = 'On'; speechToggle.style.display = 'inline-block'; liveBtn.style.display='none'; };
      recognition.onend = ()=> { recognizing = false; micStatus.textContent = 'Off'; speechToggle.style.display = 'none'; liveBtn.style.display='inline-block'; };
      recognition.onerror = (e)=> { console.warn('Speech error', e); recognizing = false; micStatus.textContent = 'Error'; speechToggle.style.display = 'none'; liveBtn.style.display='inline-block'; };

      recognition.onresult = (ev)=>{
        // build text from interim + final
        let text = '';
        for(let i = ev.resultIndex; i < ev.results.length; i++){
          text += ev.results[i][0].transcript;
        }
        // show in caption + also append small visual
        typeEl.textContent = text;
      };
    } else {
      // hide live capture button if not supported
      liveBtn.title = "Live speech captions are not supported in this browser";
      liveBtn.style.opacity = 0.9;
    }

    liveBtn.addEventListener('click', ()=>{
      if(!recognition){ alert('Live speech captions not available in this browser.'); return; }
      try{ recognition.start(); }catch(e){ console.warn(e); recognition.start(); }
    });
    speechToggle.addEventListener('click', ()=>{
      if(recognizing) recognition.stop();
      else recognition.start();
    });

    /* ------------------------------
       Small UX: 'Shop' button scroll to grid
       ------------------------------ */
    document.getElementById('startShop').addEventListener('click', ()=> {
      document.querySelector('main').scrollIntoView({behavior:'smooth'});
    });

    /* ------------------------------
       Theme button (tiny accent toggle)
       ------------------------------ */
    document.getElementById('themeBtn').addEventListener('click', ()=>{
      document.documentElement.style.setProperty('--accent', getComputedStyle(document.documentElement).getPropertyValue('--accent-2') || '#ffd166');
      document.getElementById('themeBtn').classList.add('pulse');
      setTimeout(()=> document.getElementById('themeBtn').classList.remove('pulse'), 900);
    });

    /* ------------------------------
       Keyboard accessibility: Esc to close modal
       ------------------------------ */
    document.addEventListener('keydown', e=>{
      if(e.key === 'Escape') closeModal();
    });
